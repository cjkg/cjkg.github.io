Sub fillConcat()
With Worksheets("macro")

Dim lineCount As Long
lineCount = Worksheets("raw data").Range("A" & Rows.Count).End(xlUp).row

Dim row As Long

    For row = 2 To lineCount
    
        Dim dateLength As Long
        dateLength = Len(Range("F" & row))
    
        Dim pmtLength As Long
        pmtLength = Len(Range("E" & row))
    
        Dim pmtSpace As Long
        pmtSpace = 12 - pmtLength
    
        Dim dateSpace As Long
        dateSpace = 11 - dateLength
    
        Range("H" & row).Value = Range("G" & row).Value & Range("A" & row).Value & Range("C" & row).Value & "   " & Range("D" & row).Value & Space(dateSpace) & Range("F" & row).Value & Space(pmtSpace) & Range("E" & row).Value
        Next row

End With

End Sub
